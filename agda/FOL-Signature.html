    data FOL (A : Set) : Set where
      ∅ : FOL A -- equivalent to True
      <_> : A → FOL A
      ~ : FOL A → FOL A
      _&_ : FOL A → FOL A → FOL A
      _||_ : FOL A → FOL A → FOL A
      exists : ℕ → FOL A → FOL A
      all : ℕ → FOL A → FOL A

    parser-empty : List Char → Bool
    parser-empty [] = true
    parser-empty (x ∷ rest) = false

    Parser : Set → Set
    Parser A = List Char → List (A × (List Char))

    pair-map₁ : {A₁ A₂ B : Set} → (A₁ → A₂) → A₁ × B → A₂ × B
    pair-map₁ f (a , b) = (f a) , b

    pair-map₂ : {A B₁ B₂ : Set} → (B₁ → B₂) → A × B₁ → A × B₂
    pair-map₂ f (a , b) = a , (f b)

    pair-map : {A₁ A₂ B₁ B₂ : Set} → (A₁ → A₂) → (B₁ → B₂) → A₁ × B₁ → A₂ × B₂
    pair-map f g (a , b) = (f a) , (g b)

    term : Char → Parser Char
    term c [] = []
    term c (x ∷ rest) = if ((primCharToNat c) eq (primCharToNat x)) then ((x , rest) ∷ []) else []

    alt : {A B : Set} → Parser A → Parser B → Parser (A ⊎ B)
    alt Pa Pb s = (map (pair-map₁ inj₁) (Pa s)) ++ (map (pair-map₁ inj₂) (Pb s))

    seq-flatten₁ : {A B C : Set} → A × (List (B × C)) → List ((A × B) × C)
    seq-flatten₁ (a , []) = []
    seq-flatten₁ (a , ((b , c) ∷ rest)) = ((a , b) , c) ∷ (seq-flatten₁ (a , rest))

    seq-flatten : {A B C : Set} → List (A × (List (B × C))) → List ((A × B) × C)
    seq-flatten l = foldr _++_ [] (map seq-flatten₁ l)

    seq : {A B : Set} → Parser A → Parser B → Parser (A × B)
    seq Pa Pb s = seq-flatten (map (pair-map₂ Pb) (Pa s))

    parserRange : ℕ → ℕ → Parser Char
    parserRange _   _   []         = []
    parserRange min max (x ∷ rest) = if (((primCharToNat x) ge min) and ((primCharToNat x) le max)) then ((x , rest) ∷ []) else []

    alphaLower : Parser Char
    alphaLower = parserRange 97 122

    alphaUpper : Parser Char
    alphaUpper = parserRange 65 90

    digit : Parser Char
    digit = parserRange 48 57

    parser-a : Parser Char
